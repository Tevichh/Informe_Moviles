La función estadisticas realiza un análisis estadístico de un conjunto de datos de medidas de RSSI (Received Signal Strength Indication) para diferentes puntos.

dataMedidas: Es un diccionario donde las claves representan los puntos de medición y los valores son DataFrames que contienen las mediciones de RSSI para cada punto.

def estadisticas(dataMedidas):
    calculos = []
    k = 0

    for punto, dataFrame in dataMedidas.items():
        
        data = dataFrame["RSSI"].copy()
        
    #ASEGURAR TOTALIDAD DE DATOS
        if (len(data)<=0):
            data.at[0] = -100
            
        if(len(data)==1):
            data.at[1] = data.at[0]
            
            
    #APLICACION DE ESTADISTICA
            
        distancia = k * 5 if k != 0 else 4
        k += 1 
        media = data.mean()  
        error_tipico = data.sem()  
        mediana = data.median()  
        moda = data.mode()
        moda = moda[0]
        desviacion_estandar = data.std() 
        varianza_muestra = data.var()  
        curtosis = data.kurtosis()  
        coeficiente_asimetria = data.skew() 
        rango = data.max() - data.min()  
        minimo = data.min()  
        maximo = data.max() 
        suma = data.sum()  
        cuenta = len(data) 
        nivel_confianza = 0 
        error_absoluto = 0 
        error_relativo = 0  
        
        calculos.append({
            "Punto": punto,
            "distancia metros": round(distancia,4),
            'Media': round(media, 4),
            'Error típico': round(error_tipico,4),
            'Mediana': round(mediana,4),
            'Moda': round(moda,4),
            'Desviación estándar': round(desviacion_estandar,4),
            'Var. muestra': round(varianza_muestra,4),
            'Curtosis': round(curtosis,4),
            'Coef. asimetría': round(coeficiente_asimetria,4),
            'Rango': round(rango,4),
            'Mínimo': round(minimo,4),
            'Máximo': round(maximo,4),
            'Suma': round(suma,4),
            'Cuenta': round(cuenta,4),
            '# confianza (95%)': round(nivel_confianza,4),
            'Error absoluto': round(error_absoluto,4),
            'Error relativo': error_relativo
        })       
    calculos = pd.DataFrame(calculos)

    return calculos

FUNCION GRAFICAR

La función graficarTendenciaLog toma un DataFrame de datos estadísticos, incluyendo la distancia en metros y la media de potencia, y traza un gráfico que muestra la tendencia logarítmica de la potencia en función de la distancia. Utilizando ajuste de regresión logarítmica, calcula la ecuación de la línea de tendencia y muestra el exponente de la pérdida de señal. Este gráfico proporciona una representación visual clara de cómo la potencia de la señal decae con la distancia, lo que es fundamental para comprender el comportamiento de la señal en un entorno dado.

aplica la regresión logarítmica utilizando la función np.polyfit() de NumPy para ajustar una línea de tendencia logarítmica a los datos de distancia y potencia. Esta regresión encuentra los coeficientes que mejor se ajustan a los datos en una forma logarítmica. Luego, utiliza estos coeficientes para crear una función de línea de tendencia logarítmica mediante np.poly1d(). Esta función calcula la potencia esperada en función de la distancia, lo que permite visualizar cómo la potencia de la señal varía de manera logarítmica con la distancia en el gráfico resultante.

def graficarTendenciaLog(calculos, nombre):
    x = calculos["distancia metros"]
    y = calculos["Media"]

    # Ajuste de línea de tendencia logarítmica
    coefficients = np.polyfit(np.log(x), y, 1)
    trend_line = np.poly1d(coefficients)

    # Obtener los coeficientes de la ecuación de la línea de tendencia logarítmica
    a, b = coefficients
    equation = f'y = {a:.2f} * ln(y) + ({b:.2f})'  # Formatea la ecuaciónS
    
    expPerdida = a/10
    valorN = f'n = {a:.2f}/{10} = {expPerdida :.4f}'
    
    # Gráfico de dispersión de los datos
    plt.scatter(x, y, label="Datos")

    # Graficar la línea de tendencia logarítmica
    plt.plot(x, trend_line(np.log(x)), color='green', label="Regresion Log")

    # Mostrar la ecuación de la línea de tendencia en el gráfico
    plt.annotate(equation, xy=(1.05, 0.88), xycoords='axes fraction', fontsize=12)
    plt.annotate(valorN, xy=(1.05, 0.78), xycoords='axes fraction', fontsize=12)

    # Personalización
    plt.ylabel('POTENCIA(MEDIA)')
    plt.xlabel('DISTANCIA')
    plt.title(f'{nombre}')
    plt.grid()
    plt.legend()

    # Mostrar el gráfico
    plt.show()
    return expPerdida

def cost231_Wal_Ike(dataMedidas, frecuencia:float, Prx:float):
    
    modelo = []
    
    distancias = dataMedidas["distancia metros"]
    
    fc_log = math.log10(frecuencia)

    
    for distancia in distancias:
        distancia = distancia / 1000
        d_log = math.log10(distancia)
        LodB = 42.6 + (26 * d_log) + (20 * fc_log)
        modelo.append({
            "distancia(Km)" : distancia,
            "LP(dB)" : LodB,
            "PrxCost231_Wal_Ike(dBm)": Prx - abs(LodB)
        })
    
    modelo = pd.DataFrame(modelo)
    
    #GRAFICAR
    x = modelo["distancia(Km)"]
    y = modelo ["PrxCost231_Wal_Ike(dBm)"]
    
    plt.plot(x, y, color="purple")
    
    formula = f"Prx - LodB = {Prx} - 42.6 + 26 log10(d)Km + 20 log10({frecuencia})"
    plt.annotate(formula, xy=(1.05, 0.58), xycoords='axes fraction', fontsize=12)
    plt.ylabel('POTENCIA(dBm)')
    plt.xlabel('DISTANCIA')
    plt.title("COST 231-WAL-IKE")
    plt.grid()
   
    
    return modelo

def okomuraHataModel(dataMedidas, frecuencia:float, htx:float, hrx:float, Prx:float):
    
    modelo = []
    
    distancias = dataMedidas["distancia metros"]
    
    fc_log = math.log10(frecuencia)
    htx_log = math.log10(htx)
    aHrx  = (1.1 * fc_log - 0.7) * hrx - (1.56 * fc_log - 0.8)
    
    for distancia in distancias:
        distancia = distancia / 1000
        d_log = math.log10(distancia)
        LPdB = 69.55 + (26.16 * fc_log) - (13.82 * htx_log) - aHrx + ((44.9 * d_log)  - (6.55 * htx_log * d_log ) )
        modelo.append({
            "distancia(Km)" : distancia,
            "LP(dB)" : LPdB,
            "PrxOkomuraHata(dBm)": Prx - abs(LPdB)
        })
    
    modelo = pd.DataFrame(modelo)
    
    #GRAFICAR
    x = modelo["distancia(Km)"]
    y = modelo ["PrxOkomuraHata(dBm)"]
    
    plt.plot(x, y, color="yellow")
    
    formula = f"Prx - LPdB = {Prx} - 69.55+ 26.16 log10({frecuencia})\n − 13.82 log10({ℎ𝑡𝑥}) − A(hrx) + (44.9 − 6.55 log10({ℎ𝑡𝑥}) * log10(d))"
    formula2 = f"A(hrx) = (1.1 * {fc_log:.4f} - 0.7) * {hrx} - (1.56 * {fc_log:.4f} - 0.8)\n\n={aHrx:.4f}"
    plt.annotate(formula, xy=(1.05, 0.58), xycoords='axes fraction', fontsize=12)
    plt.annotate(formula2, xy=(1.05, 0.78), xycoords='axes fraction', fontsize=12)
    plt.ylabel('POTENCIA(dBm)')
    plt.xlabel('DISTANCIA')
    plt.title("OKOMURA HATA")
    plt.grid()
   
    
    return modelo

def modelo_ericsson(dataMedidas, hb, hr, f, Prx, a0, a1, a2, a3):
    modelo = []
    
    distancias = dataMedidas["distancia metros"]
    

    # Función g(f)
    g_f = 44.49 * math.log10(f) - 4.78 * (math.log10(f))**2

    for distancia in distancias:
        L = a0 + a1 * math.log10(distancia) + a2 * math.log10(hb) + a3 * math.log10(hb) * math.log10(distancia) \
                - 3.2 * (math.log10(11.75 * hr))**2 + g_f * math.log10(distancia)
        modelo.append({
            "distancia(m)": distancia,  
            "LP(dB)": L,
            "PrxEricsson(dBm)": Prx - abs(L)  
        })
        
    modelo = pd.DataFrame(modelo)
    
    #GRAFICAR
    x = modelo["distancia(m)"]
    y = modelo ["PrxEricsson(dBm)"]
    
    plt.plot(x, y, color="red")
    
    formula = f"Prx - LPdB = {Prx} - {a0} + {a1} log(d) + {a2} log({hb}) + {a3}log({hb}) log(d) - 3.2(log(11.75 * {hr}))2 +g(f) "
    formula2 = f"g(f) = 44.49 log({f}) - 4.78(log{f})^2"
    plt.annotate(formula, xy=(1.05, 0.58), xycoords='axes fraction', fontsize=12)
    plt.annotate(formula2, xy=(1.05, 0.78), xycoords='axes fraction', fontsize=12)
    plt.ylabel('POTENCIA(dBm)')
    plt.xlabel('DISTANCIA')
    plt.title("ERISCSON")
    plt.grid()
        
    return modelo
