def estadisticas(dataMedidas):
    calculos = []
    k = 0

    for punto, dataFrame in dataMedidas.items():
        
        data = dataFrame["RSSI"].copy()
        
    #ASEGURAR TOTALIDAD DE DATOS
        if (len(data)<=0):
            data.at[0] = -100
            
        if(len(data)==1):
            data.at[1] = data.at[0]
            
            
    #APLICACION DE ESTADISTICA
            
        distancia = k * 5 if k != 0 else 4
        k += 1 
        media = data.mean()  
        error_tipico = data.sem()  
        mediana = data.median()  
        moda = data.mode()
        moda = moda[0]
        desviacion_estandar = data.std() 
        varianza_muestra = data.var()  
        curtosis = data.kurtosis()  
        coeficiente_asimetria = data.skew() 
        rango = data.max() - data.min()  
        minimo = data.min()  
        maximo = data.max() 
        suma = data.sum()  
        cuenta = len(data) 
        nivel_confianza = 0 #stats.t.interval(0.95, len(data)-1, loc=np.mean(data), scale=stats.sem(data))
        error_absoluto = 0 #abs(data - media)
        error_relativo = 0  
        
        calculos.append({
            "Punto": punto,
            "distancia_metros": distancia,
            'Media': media,
            'Error típico': error_tipico,
            'Mediana': mediana,
            'Moda': moda,
            'Desviación estándar': desviacion_estandar,
            'Varianza de la muestra': varianza_muestra,
            'Curtosis': curtosis,
            'Coeficiente de asimetría': coeficiente_asimetria,
            'Rango': rango,
            'Mínimo': minimo,
            'Máximo': maximo,
            'Suma': suma,
            'Cuenta': cuenta,
            'Nivel de confianza(95%)': nivel_confianza,
            'Error absoluto': error_absoluto,
            'Error relativo': error_relativo
        })       
    calculos = pd.DataFrame(calculos)

    return calculos

def graficarTendenciaLog(calculos, nombre):
    x = calculos["distancia_metros"]
    y = calculos["Media"]

    # Ajuste de línea de tendencia logarítmica
    coefficients = np.polyfit(np.log(x), y, 1)
    trend_line = np.poly1d(coefficients)

    # Obtener los coeficientes de la ecuación de la línea de tendencia logarítmica
    a, b = coefficients
    equation = f'y = {a:.2f} * ln(y) + ({b:.2f})'  # Formatea la ecuaciónS
    
    expPerdida = a/10
    valorN = f'n = {a:.2f}/{10} = {expPerdida :.4f}'
    
    # Gráfico de dispersión de los datos
    plt.scatter(x, y, label="Datos")

    # Graficar la línea de tendencia logarítmica
    plt.plot(x, trend_line(np.log(x)), color='green', label="Regresion Log")

    # Mostrar la ecuación de la línea de tendencia en el gráfico
    plt.annotate(equation, xy=(1.05, 0.88), xycoords='axes fraction', fontsize=12)
    plt.annotate(valorN, xy=(1.05, 0.78), xycoords='axes fraction', fontsize=12)

    # Personalización
    plt.ylabel('POTENCIA(MEDIA)')
    plt.xlabel('DISTANCIA')
    plt.title(f'{nombre}')
    plt.grid()
    plt.legend()

    # Mostrar el gráfico
    plt.show()
    return expPerdida

def okomuraHataModel(dataMedidas, frecuencia:float, htx:float, hrx:float, Prx:float):
    
    modelo = []
    
    distancias = dataMedidas["distancia_metros"]
    
    fc_log = math.log10(frecuencia)
    htx_log = math.log10(htx)
    aHrx  = (1.1 * fc_log - 0.7) * hrx - (1.56 * fc_log - 0.8)
    
    for distancia in distancias:
        distancia = distancia / 1000
        distancia
        d_log = math.log10(distancia)
        LPdB = 69.55 + (26.16 * fc_log) - (13.82 * htx_log) - aHrx + ((44.9 * d_log)  - (6.55 * htx_log * d_log ) )
        modelo.append({
            "distancia(Km)" : distancia,
            "LP(dB)" : LPdB,
            "PrxOkomuraHata(dBm)": Prx - abs(LPdB)
        })
    
    modelo = pd.DataFrame(modelo)
    
    #GRAFICAR
    x = modelo["distancia(Km)"]
    y = modelo ["PrxOkomuraHata(dBm)"]
    
    plt.plot(x, y, color="red")
    
    formula = f"Prx + LPdB = {Prx} + 69.55+ 26.16 log10({frecuencia})\n − 13.82 log10({ℎ𝑡𝑥}) − A(hrx) + (44.9 − 6.55 log10({ℎ𝑡𝑥}) * log10(d))"
    formula2 = f"A(hrx) = (1.1 * {fc_log:.4f} - 0.7) * {hrx} - (1.56 * {fc_log:.4f} - 0.8)\n\n={aHrx:.4f}"
    plt.annotate(formula, xy=(1.05, 0.58), xycoords='axes fraction', fontsize=12)
    plt.annotate(formula2, xy=(1.05, 0.78), xycoords='axes fraction', fontsize=12)
    plt.ylabel('POTENCIA(dBm)')
    plt.xlabel('DISTANCIA')
    plt.title("OKOMURA HATA")
    plt.grid()
   
    
    return modelo
